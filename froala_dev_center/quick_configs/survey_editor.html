Configuration

JavaScript Configuration 
const surveyEditorConfig = {
  // Survey-specific plugins
  pluginsEnabled: [
    'align', 'colors', 'fontSize',
    'paragraphFormat', 'lists',
    'link', 'image', 'table',
    'charCounter', 'wordCounter',
    'specialCharacters', 'fullscreen'
  ],

  // Survey toolbar - focused on structure
  toolbarButtons: {
    'moreText': {
      'buttons': ['bold', 'italic', 'underline',
                  'fontSize', 'textColor', 'clearFormatting']
    },
    'moreParagraph': {
      'buttons': ['alignLeft', 'alignCenter', 'alignRight',
                  'formatOL', 'formatUL', 'paragraphFormat',
                  'outdent', 'indent']
    },
    'moreRich': {
      'buttons': ['insertLink', 'insertImage', 'insertTable',
                  'specialCharacters', 'insertHR']
    },
    'moreMisc': {
      'buttons': ['undo', 'redo', 'fullscreen', 'selectAll']
    }
  },

  // Height settings
  heightMin: 400,

  // Placeholder
  placeholderText: 'Create your survey questions...',

  // Character counter
  charCounterCount: true,
  wordCounterMax: -1,

  // Paragraph formats for survey structure
  paragraphFormat: {
    'N': 'Normal',
    'H2': 'Section Title',
    'H3': 'Question',
    'H4': 'Sub-question'
  },

  // Font sizes for emphasis
  fontSize: ['12', '14', '16', '18', '20', '24'],

  // Image settings - for visual aids in surveys
  imageEditButtons: ['imageReplace', 'imageAlign', 'imageRemove', 'imageSize'],
  imageInsertButtons: ['imageBack', '|', 'imageUpload', 'imageByURL'],
  imageMaxSize: 5 * 1024 * 1024, // 5MB
  imageAllowedTypes: ['jpeg', 'jpg', 'png', 'gif'],

  // Table settings - useful for matrix questions
  tableEditButtons: ['tableHeader', 'tableRemove', '|', 'tableRows', 'tableColumns',
                     'tableStyle', '-', 'tableCells', 'tableCellBackground',
                     'tableCellVerticalAlign', 'tableCellHorizontalAlign'],
  tableStyles: {
    'fr-table-bordered': 'Bordered',
    'fr-table-striped': 'Striped',
    'fr-table-compact': 'Compact'
  },
  tableDefaultWidth: '100%',

  // Link settings
  linkEditButtons: ['linkOpen', 'linkEdit', 'linkRemove'],
  linkInsertButtons: ['linkBack'],
  linkAlwaysBlank: true,
  linkAutoPrefix: 'https://',

  // Special characters for survey formatting
  specialCharactersSet: ['©', '®', '™', '§', '¶', '•', '→', '✓', '✗',
                         '⭐', '❤️', '👍', '👎'],

  // List advanced types for different question formats
  listAdvancedTypes: true,

  // Clean paste for survey content
  pastePlain: false,
  pasteDeniedTags: ['script', 'style'],
  pasteDeniedAttrs: ['style', 'class'],

  // Disable quick insert for cleaner interface
  quickInsertEnabled: false,

  // Toolbar not sticky for better form flow
  toolbarSticky: false,

  // Enter behavior
  enter: FroalaEditor.ENTER_P,

  // Events
  events: {
    'initialized': function () {
      console.log('Survey Editor initialized');
    },
    'contentChanged': function () {
      console.log('Survey content updated');
      // Auto-save survey draft
    }
  }
};
