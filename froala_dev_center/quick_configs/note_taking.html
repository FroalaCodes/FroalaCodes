Froala Configuration for Notes

This demo uses the following Froala Editor configuration optimized for note-taking apps:

JavaScript Configuration 
new FroalaEditor('#froala-editor', {
      key: 'YOUR_LICENSE_KEY',
      
      // Plugins for essential note-taking
      pluginsEnabled: [
        'align', 'colors', 'fontFamily', 'fontSize',
        'paragraphFormat', 'lists', 'link', 'image',
        'table', 'quote', 'specialCharacters', 'wordCounter',
        'quickInsert', 'save'
      ],
    
      // Clean, minimal toolbar
      toolbarButtons: {
        'moreText': {
          'buttons': ['bold', 'italic', 'underline', 'strikeThrough', 'textColor', 'backgroundColor', 'clearFormatting']
        },
        'moreParagraph': {
          'buttons': ['alignLeft', 'alignCenter', 'formatOLSimple', 'formatUL', 'paragraphFormat', 'quote']
        },
        'moreRich': {
          'buttons': ['insertLink', 'insertImage', 'insertTable', 'specialCharacters', 'insertHR']
        },
        'moreMisc': {
          'buttons': ['undo', 'redo', 'selectAll']
        }
      },
      
      // Editor settings
      heightMin: 600,
      placeholderText: 'Start typing your note...',
      
      // Quick insert for rapid content creation
      quickInsertEnabled: true,
      quickInsertButtons: ['image', 'table', 'ul', 'ol', 'hr'],
      
      // Auto-save configuration
      saveInterval: 2000,
      saveMethod: 'POST',
      
      // Typography tailored for reading notes
      fontFamily: {
        'Inter,sans-serif': 'Inter',
        'Arial,Helvetica,sans-serif': 'Arial',
        'Georgia,serif': 'Georgia',
        'Courier New,Courier,monospace': 'Monospace'
      },
      
      // Clean paste to avoid messy formatting from other sources
      pastePlain: true,
      
      // Event handlers
      events: {
        'focus': function () {
          console.log('Editor focused');
        },
        'blur': function () {
          console.log('Editor saved');
        }
      }
    });
